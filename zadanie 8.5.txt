select i.summary, i.description, l.NAME
from ISSUES i
join ISSUESLISTS l on i.ISSUESLISTS_ID = l.ID;

select u.LASTNAME, i.summary, i.description
from ISSUES i
join USERS u on i.USER_ID_ASSIGNEDTO = u.id
order by u.LASTNAME, i.id;

select u.FIRSTNAME, u.LASTNAME, count(*) as posts_number
from USERS u
join POSTS p on u.ID = p.USER_ID
group by u.ID;
